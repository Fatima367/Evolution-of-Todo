📊 View Logs:

🔧 Useful Commands:
  # Port forward to backend
  kubectl port-forward service/{{ include "todoboard.fullname" . }}-backend {{ .Values.backend.service.port }}:{{ .Values.backend.service.port }} -n {{ .Release.Namespace }}

  # Port forward to frontend
  kubectl port-forward service/{{ include "todoboard.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }} -n {{ .Release.Namespace }}

  # Connect to database
  kubectl exec -it deployment/{{ include "todoboard.fullname" . }}-postgres -n {{ .Release.Namespace }} -- psql -U {{ .Values.postgresql.auth.username }} -d {{ .Values.postgresql.auth.database }}

  # View deployment details
  helm status {{ .Release.Name }} -n {{ .Release.Namespace }}

  # Upgrade the deployment
  helm upgrade {{ .Release.Name }} . -f values-minikube.yaml -n {{ .Release.Namespace }}

  # Rollback to previous version
  helm rollback {{ .Release.Name }} -n {{ .Release.Namespace }}

  # Uninstall the deployment
  helm uninstall {{ .Release.Name }} -n {{ .Release.Namespace }}

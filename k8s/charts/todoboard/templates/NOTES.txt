{{- else if eq .Values.frontend.service.type "ClusterIP" }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "todoboard.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app=todoboard-frontend" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3000:{{ .Values.frontend.service.port }}
